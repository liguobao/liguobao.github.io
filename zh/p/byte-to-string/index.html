	private static byte[] StringToBytes(string str)
	{
	    if (string.IsNullOrEmpty(str))
	        return null;
	    byte[] bytes = new byte[str.Length / 2];
	    for (int i = 0; i < str.Length; i += 2)
	    {
	        bytes[i / 2] = Convert.ToByte("0x" + str[i] + str[i + 1], 16);

	    return bytes;
